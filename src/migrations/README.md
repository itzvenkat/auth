# Migrations in this directory are generated and managed by the TypeORM CLI.
#
# Commands:
#   npm run migration:generate -- src/migrations/MigrationName   (auto-diff entities → SQL)
#   npm run migration:run                                          (apply all pending)
#   npm run migration:revert                                       (undo last applied)
#   npm run migration:show                                         (list applied / pending)
#   npm run migration:create -- src/migrations/EmptyMigration      (blank template)
#
# Naming convention: <timestamp>-<PascalCaseName>.ts
#   e.g. 1708600000000-CreateUsersTable.ts
#
# Never edit a migration that has already been applied to a shared/production DB.
# Create a new migration instead.
